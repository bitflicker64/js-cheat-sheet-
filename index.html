While Loop

    i *= 2;

Do While Loop

do {
    i *= 2;
    document.write(i + ", ");
} while (i < 100);

Break

    if (i == 5) break;
    document.write(i + ", ");
}

Continue

    if (i == 5) continue;
    document.write(i + ", ");
}

Promises

function sum(a, b) {
    return Promise(function(resolve, reject) {
        setTimeout(function() {
            if (typeof a !== "number" || typeof b !== "number") {
                return reject(new TypeError("Inputs must be numbers"));
            }
            resolve(a + b);
        }, 1000);
    });
}

var myPromise = sum(10, 5);
myPromise.then(function(result) {
    document.write(result);
}).catch(function(err) {
    console.error(err);
});
